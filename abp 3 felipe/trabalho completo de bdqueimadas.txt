-- 1. Criação da sequência para o campo 'id'
CREATE SEQUENCE IF NOT EXISTS risco_id_seq;

-- 2. Criação da tabela 'Risco' com todos os tipos e restrições corretos
CREATE TABLE IF NOT EXISTS Risco (
    id INTEGER PRIMARY KEY DEFAULT nextval('risco_id_seq'), -- ID autoincremental
    data DATE,                                              -- Data do registro
    geometria GEOMETRY(POINT, 4326),                        -- Geometria com SRID 4326
    estado_id INTEGER,                                      -- Referência ao estado
    bioma_id INTEGER,                                       -- Referência ao bioma
    risco_fogo NUMERIC(10,2),                               -- Risco com até 2 casas decimais
    FOREIGN KEY (estado_id) REFERENCES Estados(id_estado),  -- FK para Estados
    FOREIGN KEY (bioma_id) REFERENCES Bioma(id)             -- FK para Bioma
);

-- Cria uma view chamada 'vw_risco_bioma_cerrado'
CREATE VIEW vw_risco_bioma_cerrado AS
SELECT
    r.id,                                   -- ID do risco
    r.data,                                 -- Data do registro
    r.geometria,                            -- Seleciona a geometria sem conversão para texto
    r.risco_fogo,                           -- Valor de risco de fogo
    b.bioma AS bioma                         -- Nome do bioma relacionado
FROM Risco r
JOIN Bioma b ON r.bioma_id = b.id          -- Junta com a tabela Bioma usando bioma_id
WHERE b.bioma = 'Cerrado';                  -- Filtra registros do bioma "Cerrado"

-- Seleciona o dia e o maior valor de risco de fogo registrado nesse dia
SELECT 
    data::DATE AS dia,                      -- Converte a data para apenas a parte da data (sem hora)
    MAX(risco_fogo) AS maior_risco          -- Encontra o maior risco de fogo para esse dia
FROM risco
GROUP BY dia                                -- Agrupa os dados por dia
ORDER BY maior_risco DESC                   -- Ordena do maior para o menor risco
LIMIT 1;                                    -- Retorna apenas o dia com o maior risco registrado


-- Seleciona o dia com o maior valor de risco de fogo, junto com o nome do estado e bioma
SELECT 
    r.data::DATE AS dia,                    -- Converte a data para apenas a parte da data (sem hora)
    r.risco_fogo AS maior_risco,            -- Valor do risco de fogo
    e.estado AS nome_estado,                -- Nome do estado (tabela Estados)
    b.bioma AS nome_bioma                   -- Nome do bioma (tabela Bioma)
FROM risco r
JOIN estados e ON r.estado_id = e.id_estado -- Relaciona com a tabela de Estados
JOIN bioma b ON r.bioma_id = b.id           -- Relaciona com a tabela de Biomas
ORDER BY r.risco_fogo DESC                  -- Ordena do maior para o menor risco
LIMIT 1;                                    -- Retorna apenas o registro com maior risco

-- Seleciona o registro completo com o maior valor de risco de fogo
SELECT 
    id,                                      -- ID do registro
    data,                                    -- Data e hora do registro
    risco_fogo,                              -- Valor de risco de fogo
    estado_id,                               -- ID do estado (relaciona com a tabela Estados)
    bioma_id                                 -- ID do bioma (relaciona com a tabela Bioma)
FROM risco
ORDER BY risco_fogo DESC                     -- Ordena do maior para o menor risco
LIMIT 1;    



-- Seleciona o dia e o maior valor de risco de fogo registrado nesse dia
SELECT

    MAX(risco_fogo) AS maior_risco          -- Encontra o maior risco de fogo para esse dia

ORDER BY maior_risco DESC                   -- Ordena do maior para o menor risco

-- Seleciona o dia e o maior valor de risco de fogo registrado no banco

SELECT 
    r.data::DATE AS dia,                    -- Apenas a parte da data
    MAX(r.risco_fogo) AS maior_risco,       -- Maior risco de fogo
    r.geometria                             -- Geometria do ponto (visualizável no pgAdmin)
FROM risco r
GROUP BY r.data::DATE, r.geometria
ORDER BY maior_risco DESC
LIMIT 5;


-- Seleciona o maior risco de fogo registrado em cada estado, com data e nome do estado
SELECT DISTINCT ON (e.estado)  -- Garante um único resultado por estado
    e.estado AS nome_estado,   -- Nome do estado
    r.data::DATE AS dia,       -- Dia do registro (sem hora)
    r.risco_fogo AS maior_risco
FROM risco r
JOIN estados e ON r.estado_id = e.id_estado
ORDER BY e.estado, r.risco_fogo DESC;


-- Seleciona o maior risco de fogo registrado em cada bioma, com data e nome do bioma
SELECT DISTINCT ON (b.bioma)  -- Garante um único resultado por bioma
    b.bioma AS nome_bioma,    -- Nome do bioma
    r.data::DATE AS dia,      -- Dia do registro
    r.risco_fogo AS maior_risco
FROM risco r
JOIN bioma b ON r.bioma_id = b.id
ORDER BY b.bioma, r.risco_fogo DESC;

-- Seleciona o maior risco de fogo registrado entre todos os biomas
SELECT 
    b.bioma AS nome_bioma,         -- Nome do bioma
    r.data::DATE AS dia,           -- Dia do registro
    r.risco_fogo AS maior_risco    -- Valor do risco de fogo
FROM risco r
JOIN bioma b ON r.bioma_id = b.id
ORDER BY r.risco_fogo DESC         -- Ordena do maior para o menor risco
LIMIT 1;                           -- Retorna apenas o maior registro

--Qual mês tem maior risco de fogo médio:


SELECT 
  mes, 
  ROUND(AVG(valor), 1) AS media_geral
FROM mes_fogo
WHERE tipo = 'media'
GROUP BY mes
ORDER BY media_geral DESC
LIMIT 1;


--Maior e menor média mensal

-- Seleciona os 5 meses com maior média de risco de fogo (valores do tipo 'media')
SELECT 
  mes, 
  ROUND(AVG(valor), 1) AS media_geral  -- Calcula a média e arredonda para 1 casa decimal
FROM mes_fogo
WHERE tipo = 'media'                   -- Filtra apenas os valores do tipo 'media'
GROUP BY mes                           -- Agrupa por mês
ORDER BY media_geral DESC              -- Ordena da maior média para a menor
LIMIT 5;                               -- Mostra só os 5 primeiros resultados


-- Consulta que retorna o mês com o maior valor máximo de risco de fogo
SELECT 
  mes, 
  valor AS valor_maximo
FROM mes_fogo
WHERE tipo = 'max'         -- Considera apenas os valores máximos
ORDER BY valor DESC        -- Ordena do maior para o menor
LIMIT 1;                   -- Retorna só o maior valor (primeiro da lista)

-- Consulta que retorna o mês com o menor valor mínimo de risco de fogo
SELECT 
  mes, 
  valor AS valor_minimo
FROM mes_fogo
WHERE tipo = 'min'         -- Considera apenas os valores mínimos
ORDER BY valor ASC         -- Ordena do menor para o maior
LIMIT 10;                   -- Retorna só o menor valor (primeiro da lista)


-- Consulta que retorna a média de risco de fogo por mês, arredondada para 1 casa decimal
--Se quiser a mesma consulta para os tipos 'max' ou 'min', é só trocar essa linha: WHERE tipo = 'media'

SELECT 
  mes, 
  ROUND(AVG(valor), 1) AS media_geral         -- Calcula a média de risco por mês e arredonda
FROM mes_fogo
WHERE tipo = 'media'                           -- Considera apenas os registros do tipo 'media'
GROUP BY mes                                   -- Agrupa os dados por mês
ORDER BY 
  CASE                                         -- Ordena os meses na sequência correta do ano
    WHEN mes = 'janeiro' THEN 1
    WHEN mes = 'fevereiro' THEN 2
    WHEN mes = 'março' THEN 3
    WHEN mes = 'abril' THEN 4
    WHEN mes = 'maio' THEN 5
    WHEN mes = 'junho' THEN 6
    WHEN mes = 'julho' THEN 7
    WHEN mes = 'agosto' THEN 8
    WHEN mes = 'setembro' THEN 9
    WHEN mes = 'outubro' THEN 10
    WHEN mes = 'novembro' THEN 11
    WHEN mes = 'dezembro' THEN 12
  END;

    id SERIAL PRIMARY KEY,                         -- ID autoincrementável
    estado_id INT REFERENCES Estados(id_estado),   -- FK para o estado (tabela Estados)
    bioma_id INT REFERENCES Bioma(id),             -- FK para o bioma (tabela Bioma)
   data_pas DATE NOT NULL,                         -- Data do foco
    risco DECIMAL(3,2),                            -- Índice de risco (0.00 a 1.00)
    frp DECIMAL(7,2),                              -- Potência radiativa do fogo (0.1 a 5000)
    geom GEOMETRY(Point, 4326),                    -- Coordenadas do foco (WGS 84)
    FOREIGN KEY (estado_id) REFERENCES Estados(id_estado),
    FOREIGN KEY (bioma_id) REFERENCES Bioma(id)
	);
